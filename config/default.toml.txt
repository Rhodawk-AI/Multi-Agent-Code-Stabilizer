# ─── Code Property Graph — Gap 1 (Joern integration) ────────────────────────
#
# Joern builds a full Code Property Graph (CPG) combining:
#   • AST  — abstract syntax tree
#   • CFG  — control flow graph
#   • PDG  — program dependence graph (data + control flow)
#
# The CPG enables three graph types used for causal context selection:
#   1. Call graph       → which functions CALL the buggy function (depth 1–3)
#   2. Data flow graph  → which variables and values FLOW INTO the crash site
#   3. Type flow graph  → which callers VIOLATE the type contract (no null guard)
#
# Together these replace vector similarity for context selection in FixerAgent:
# instead of "code that looks similar", the agent receives "code that CAUSED
# the bug", including cross-file, cross-service causal chains that embeddings
# cannot detect.
#
# Quick start:
#   export JOERN_REPO_PATH=/path/to/repo-being-audited
#   docker-compose up joern
#   # CPG is built automatically on first run (5–30 min for 10M lines)
#
# All settings here map 1-to-1 to StabilizerConfig fields (config/loader.py).
# They can all be overridden via environment variables (JOERN_URL, etc.).
[cpg]

# Master switch.  Set false to skip Joern entirely and fall back to the
# hybrid BM25+dense retrieval path (same quality as before Gap 1).
enabled                = true

# Joern HTTP server address.  Matches the docker-compose service port.
# Override with: JOERN_URL=http://joern-host:8080
joern_url              = "http://localhost:8080"

# Absolute host path of the repository to audit.  The docker-compose.yml
# mounts this at /repo inside the Joern container (read-only).
# Override with: JOERN_REPO_PATH=/absolute/path/to/repo
# Leave blank to default to StabilizerConfig.repo_root.
joern_repo_path        = ""

# Joern project name.  Re-using the same name across runs avoids a full
# CPG rebuild — Joern loads the persisted workspace from joern_workspace volume.

# Functions in the CPG impact set above this count trigger human review

# ─── Synthesis Agent — Gap 2 (Cross-domain compound finding detection) ───────
#
# Three AuditorAgent instances run in parallel — SECURITY, ARCHITECTURE,
# STANDARDS — all calling the same LLM with different prompts.  Without
# synthesis, this produces duplicate findings and misses cross-domain bugs.
#
# The SynthesisAgent sits between the audit phase and the consensus phase:
#
#   AuditorAgent(SECURITY)     ─┐
#   AuditorAgent(ARCHITECTURE) ─┼──→  SynthesisAgent ──→ ConsensusEngine
#   AuditorAgent(STANDARDS)    ─┘
#
# It performs two operations:
#
#   1. DEDUPLICATION — fingerprint + LLM semantic pass removes findings
#      that describe the same root defect from different angles.
#
#   2. COMPOUND FINDING DETECTION — identifies vulnerabilities that span
#      domain boundaries.  A race condition (ARCHITECTURE) that enables a
#      privilege escalation (SECURITY) is a CompoundFinding with amplification
#      ≥ 3.0.  No commercial tool finds these.
#
# The synthesis_model should be a DIFFERENT model family from the auditors
# (which default to primary_model).  Using a different family ensures the
# synthesis pass brings genuinely fresh cross-domain reasoning.
#
# Quick start:
#   Set synthesis_model in [synthesis] to a different family than primary.
#   Leave synthesis_enabled = true (default) to activate immediately.
#
[synthesis]

# Master switch.  Set false to skip synthesis and run the original three-
# auditor pipeline unchanged (same behaviour as before Gap 2).
enabled                = true

# Fingerprint + LLM semantic deduplication of same-root-defect findings.
# Disabling produces duplicate findings in the final issue list.
dedup_enabled          = true

# Cross-domain compound finding detection.
# Set false to only run deduplication without compound analysis.
compound_enabled       = true

# Model used for synthesis reasoning.  MUST be a different family from
# the primary auditors for genuine independence.
#   Example: if primary = Qwen2.5-Coder (Alibaba), use:
#     synthesis_model = "openrouter/deepseek/deepseek-coder-v2-0724"
#   or for cloud deployment:
#     synthesis_model = "claude-opus-4-20250514"
# Defaults to critical_fix_model when blank.
# Override with: RHODAWK_SYNTHESIS_MODEL=<model_id>
synthesis_model        = ""

# Maximum compound findings emitted per synthesis pass.
# Compound findings are high-value but expensive to fix — cap prevents
# overwhelming the fix queue on large codebases.
max_compound_findings  = 20

# ─── Gap 5: Multi-Intelligence / Adversarial Ensemble ────────────────────────
#
# The audit report identified that running three AuditorAgents with different
# prompts on the SAME model is parallel sampling, not adversarial intelligence.
# Gap 5 replaces the single-fixer path with a full adversarial BoBN ensemble:
#
#   Bug Report / Changed Diff
#            │
#            ▼
#     Localization Agent     ← CPG causal slice (Gap 1 integration)
#            │
#     ┌──────┴──────┐
#     │             │
#   Fixer A      Fixer B     ← Qwen2.5-Coder-32B + DeepSeek-Coder-V2-16B
#  (temp 0.2/   (temp 0.3/   ← different model FAMILIES → uncorrelated patches
#   0.4/0.6)     0.7)
#     │             │
#     └──────┬──────┘
#            │
#     Adversarial Critic     ← Llama-3.3-70B (Meta family ≠ Alibaba/DeepSeek)
#     (explicitly attacks     ← job: find ways to BREAK each candidate
#      every candidate)
#            │
#     Composite ranking      ← test_score×0.6 + robustness×0.3 + minimal×0.1
#            │
#     Formal Gate (Z3/CBMC)  ← existing FormalVerifierAgent
#            │
#     Test Generator         ← existing TestGeneratorAgent + mutation
#            │
#     Commit
#
# vLLM server setup (reference):
#   # Fixer A — primary (existing)
#   vllm serve Qwen/Qwen2.5-Coder-32B-Instruct --port 8000
#   # Fixer B — secondary (new GPU partition)
#   vllm serve deepseek-ai/DeepSeek-Coder-V2-Lite-Instruct --port 8001
#   # Critic  — 70B (4×A100 or OpenRouter)
#   vllm serve meta-llama/Llama-3.3-70B-Instruct --port 8002
#
# All settings here map 1-to-1 to StabilizerConfig fields (config/loader.py).
# They can be overridden via environment variables listed in each comment.
